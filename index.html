Docker笔记

发表于 2023-04-06 更新于 2023-10-17 分类于 编程 ， docker

常用命令

docker

docker-compose

docker-compose -h
docker-compose up # -d(后台) --rebuild(重编)
docker-compose down/stop/start
docker-compose -f xx.yaml up/down/stop/start# 指定文件

容器

docker ps # -a(包括已停止的)
docker start/stop $container
docker exec -it $container bash # 进入容器命令行

镜像

docker images
docker rmi $(docker images | grep "none" | awk '{print $3}') 

# 直接删除带none的镜像，会报错，提示先停止容器。
docker stop $(docker ps -a | grep "Exited" | awk '{print $1 }') //停止容器
docker rm $(docker ps -a | grep "Exited" | awk '{print $1 }') //删除容器
docker rmi $(docker images | grep "none" | awk '{print $3}') //删除镜像

  --name=code-server \
  -e PUID=1000 \
  -e PGID=1000 \	
  -e TZ=Europe/London \
  -e PASSWORD=tyty.1314 \
  -e SUDO_PASSWORD=Tyty.1314 \

  -p 8000:8443 \

  lscr.io/linuxserver/code-server:latest
